CREATE TABLE LEGACY_POLICY (
  POLICY_ID        VARCHAR(20),
  POLICY_NUMBER    VARCHAR(30),
  POLICY_TYPE      VARCHAR(20), -- AUTO, PROPERTY, LIABILITY
  EFFECTIVE_DATE   DATE,
  EXPIRY_DATE      DATE
);

CREATE TABLE LEGACY_INSURED (
  INSURED_ID   VARCHAR(20),
  FULL_NAME    VARCHAR(100),
  ADDRESS      VARCHAR(200)
);

CREATE TABLE LEGACY_CLAIM (
  CLAIM_ID        VARCHAR(20),
  POLICY_ID       VARCHAR(20),
  INSURED_ID      VARCHAR(20),
  LOSS_DATE       DATE,
  REPORT_DATE     DATE,
  CLAIM_STATUS    VARCHAR(20), -- OPEN, CLOSED, REOPENED
  CLAIM_TYPE      VARCHAR(20)  -- AUTO, PROPERTY, LIABILITY
);

CREATE TABLE LEGACY_EXPOSURE (
  EXPOSURE_ID   VARCHAR(20),
  CLAIM_ID      VARCHAR(20),
  COVERAGE_TYPE VARCHAR(30),
  DEDUCTIBLE    DECIMAL(10,2)
);

CREATE TABLE LEGACY_RESERVE (
  RESERVE_ID    VARCHAR(20),
  CLAIM_ID      VARCHAR(20),
  AMOUNT        DECIMAL(12,2)
);

CREATE TABLE LEGACY_PAYMENT (
  PAYMENT_ID    VARCHAR(20),
  CLAIM_ID      VARCHAR(20),
  PAID_AMOUNT   DECIMAL(12,2),
  PAYMENT_DATE  DATE
);
